卸载原有mysql


[root@hadoop1 home]# rpm -qa|grep mysql

[root@hadoop1 home]# rpm -e --nodeps mysql-libs-5.1.66-2.el6_3.x86_64

[root@hadoop1 home]#

mysql配置

安装完成后会有一个mysqld的服务，用来启动启动mysql，第一次启动mysql会自动进行配置


[root@hadoop1 home]# service mysqld start

检查mysqld是否开机启动


[root@hadoop1 home]# chkconfig --list | grep mysqld
mysqld          0:off   1:off   2:off   3:off   4:off   5:off   6:off

设置mysqld开机启动


[root@hadoop1 home]# chkconfig mysqld on

给mysqld的root账户设置密码


[root@hadoop1 home]# mysqladmin -u root password 'root'

登录mysql


[root@hadoop1 home]# mysql -u root -p

mysqld的配置文件


[root@hadoop1 etc]# cat my.cnf
[mysqld]
datadir=/var/lib/mysql
socket=/var/lib/mysql/mysql.sock
user=mysql
# Disabling symbolic-links is recommended to prevent assorted security risks
symbolic-links=0

[mysqld_safe]
log-error=/var/log/mysqld.log
pid-file=/var/run/mysqld/mysqld.pid
