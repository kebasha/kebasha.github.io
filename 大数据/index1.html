安装、登录命令


--登录
#login neo -p localhost

--关机
#shutdown -h now

--关机
#halt

--重启
#reboot

--和cp相似，一般用于makefile中，可以控制文件的属性
#install -D x /a/b/c

--挂载
#mount /dev/sdc1 /mnt/usbhd1

--卸载
#umount /mnt/usbhd1

--更改shell
#chsh -s /bin/zsh

--退出
#exit

--显示最后登录系统的N条记录
#last -10

文件处理命令


--查看文件类型
#file temp.txt

--创建目录
#mkdir -p a/b

--创建文件
#touch filename

--显示以d开头的文件中包含test的行
#grep 'test' d*

--拷贝文件，也可以创建指定大小的空文件
#dd if=sfile of=dfile

--查找
#find [/home] -name april* 没有指定目录则在当前目录下查找以april开始的文件
#find -name april* fprint file 查找以april开始的文件，并把结果输出到file中
#find -name ap* -o -name may* 查找以ap或may开头的文件
#find /home -mtime -2 查询最近两天内改动过的文件
#find /home -atime -1 查询1天之内被存取过的文件
#find /home -mmin +60 查询60分钟前改动过的文件
#find /home -newer tmp.txt 查找更新时间比tmp.txt近的文件或目录
#find /home -nouser[-nogroup] 查找不属于本地用户、组的文件或目录
#find /home -empty 查找大小为0的文件或目录
#find /home -size +(-)512k 查找大于、小于512k的文件

--移动文件，可以重命名
#mv filename newfilename

--列出文件
#ls -lh
#ls -lrt

--比较
#diff file1 file2

--显示文件内容、创建文件、合并文件
#cat filename
#cat > filename
#cat file1 file2 > file

--软链接、硬链接
#ln -s tomcat /apache/bin/tomcat

系统安全相关


--修改密码
#passwd user 设置密码
#passwd -l user 锁定密码，即用户不用修改
#passwd -d user 清除密码

--切换用户
#su user
#su - user 切换时执行环境.profile文件

--设置默认权限
#umask

#chgrp

#chmod u+x file 给file的属主增加执行权限
#chmod 751 file 为所有用户分配读权限
#chmod u=rwx,g=rx,o=x file
#chmod =r file 为所有用户分配读权限
#chmod -R u+r directory 递归地给directory目录下所有文件和子目录的属主分配读的权限

--改变文件的拥有者和群组
#chown mail:mail file
#chown -R mail:mail directory

#chattr +a 文件只能增加数据，不能删除修改
#chattr +i 不能删除修改增加连接等操作
#chattr +u 若删除，还在磁盘中，可以恢复文件

#sudo

--查看进程
#ps -u root 显示指定用户进程信息
#ps -ef 显示所有进程信息，连同命令行
#ps -ef|grep node
#ps aux 列出目前所有的正在内存当中的程序
#ps aux | egrep '(cron|syslog)' 找出与 cron 与 syslog 这两个服务有关的 PID 号码(egrep支持正则表达式)

#who

其他命令



--v显示信息
#tar -cvf xxx.tar /etc 创建包
#tar -zcvf xxx.tar.gz /etc 创建包并压缩
#tar -tvf xxx.tar 查看包内容
#tar -ztvf xxx.tar.gz 查看压缩包内容
#tar -xvf xxx.tar 解压包
#tar -zxvf xxx.tar.gz 解压压缩包
#tar -xvf xxx.tar etc/passwd //只解压其中的一个文件

#zip -r xxx.zip abc.txt dir1 把一个文件和目录压缩成xxx.zip
#unzip xxx.zip 解压缩
#unzip -v large.zip 查看压缩包里的内容
#unzip -t xxx.zip 验证压缩包是否完整

#gzip xxx.txt 将xxx.txt压缩为xxx.txt.gx，原文件会删除
#gunzip xxx.txt.gz 解压缩文件，解压完后压缩包会删除
#gzip -c xxx.txt > /home/xxx.gz 保留原文件，同时指定压缩包位置
#gunzip -c /home/xxx.gz > ./xxx.txt

#unarj
