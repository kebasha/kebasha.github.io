mongoose连接MongoDB的坑
2016-04-04 20:47:24

mongoose连接MongoDB

/*在models/user.js文件中*/
var mongoose = require('mongoose'),
    Schema = mongoose.Schema,
    ObjectId = Schema.ObjectId;

var UserSchema = new Schema({
    username: String,
    passwd: String
});

module.exports = mongoose.model('User', UserSchema);



/*在app.js文件中*/
var mongoose = require('mongoose');
var User = require('./models/user');
mongoose.connect('mongodb://192.168.3.110/angularweb');

app.use('/', function(req, res){
  User.find(function(err, docs){
    if(err){
      console.log(err);
    }
    console.log(docs);
    res.render('test', {title: 'MongoDB1', data: docs});
  });
});

今天在用mongoose连接MongoDB时，使用上面的方法试了半天，一直都查不出结果，也没有错误信息。可以肯定的是已经连接上数据库，并且数据库也有数据。实在想不明白为什么，后来写了一个保存的方法试了下，发现数据库中多了个users的collection，这才明白，原来程序自动在collection名为user的后面加上了s，所以一直都不是查询的原本的user。为了避免自动加上s，只需要进行如下改动：


/*在user.js中的最后一行，加上一个参数，如果没有最后这个参数，默认会以User加上s为数据库的collection进行使用，而第三个参数才对应的数据库的collection.不要太相信网上搜索到的示例，也许别人根本就没自己测试过。*/
module.exports = mongoose.model('User', UserSchema, 'user');
