javascript面向对象(封装)
2016-03-28 23:05:38

生成对象的原始模式


var Object = {
  name : '',
  color : ''
}

var obj1 = {};
obj1.name = "啊黄";
obj1.color = "黄色";

var obj2 = {};
obj2.name = "啊黑";
obj2.color = "黑色";

原始模式的改进


function Object(name, color){
  return {
    name:name,
    color:color
  }
}

var obj1 = Object("啊黄","黄色");
var obj2 = Object("啊黑","黑色");

构造函数模式


function Object(name,color){
  this.name = name;
  this.color = color;
}

var obj1 = new Object("啊黄","黄色");
var obj2 = new Object("啊黑","黑色");

/*这时obj1和obj2会自动含有一个constructor属性，指向它们的构造函数*/
alert(obj1.constructor == Object);//true
alert(obj2.constructor == Object);//true

构造函数模式的问题


function Object(name,color){
  this.name = name;
  this.color = color;
  this.type = "this is type!";
}

var obj1 = new Object("啊黄","黄色");
var obj2 = new Object("啊黑","黑色");

alert(obj1.type == obj2.type); //false

表面上好像没什么问题，但是实际上这样做，有一个很大的弊端。那就是对于每一个实例对象，type属性都是一模一样的内容，每一次生成一个实例，都必须为重复的内容，多占用一些内存。这样既不环保，也缺乏效率。

Prototype模式

Javascript规定，每一个构造函数都有一个prototype属性，指向另一个对象。这个对象的所有属性和方法，都会被构造函数的实例继承。这意味着，我们可以把那些不变的属性和方法，直接定义在prototype对象上。


function Object(name,color){
  this.name = name;
  this.color = color;
}
Object.prototype.type = "this is type!";

var obj1 = new Object("啊黄","黄色");
var obj2 = new Object("啊黑","黑色");

alert(obj1.type == obj2.type); //true
